The resolveMotivation function receives two parameters:
- pct = avgDiff / avgRef
- userGoal = "deficit" | "maintain" | "surplus"

GOAL: deficit (current rules, no change)

GOAL: maintain
if (pct <= -0.50) → Alto deficit
  title: "Deficit calorico"
  emoji: "📋"
  message: "Deficit alto, controlla di aver loggato tutti i pasti"

if (pct > -0.50 && pct <= -0.05) → Sotto media
  title: "Sotto la media"
  emoji: "🏃"
  message: "Stai mangiando un po' meno del necessario"

if (pct > -0.05 && pct < 0.05) → Nella media ✅ GOOD STATE
  title: "Nella media"
  emoji: "🌟"
  message: "Perfetto! Stai mantenendo il tuo obiettivo"

if (pct >= 0.05 && pct < 0.50) → Sopra media
  title: "Sopra la media"
  emoji: "🏃"
  message: "Essere consapevole è già un passo, continuiamo"

if (pct >= 0.50) → Alto surplus
  title: "Surplus calorico"
  emoji: "🏃"
  message: "Succede, continua e rientriamo nell'obiettivo"

GOAL: surplus (mirror of deficit)

  title: "Surplus calorico"

  title: "Surplus calorico"
  emoji: "🌟"
  message: "Ottimo! Stai andando verso il tuo obiettivo"

  title: "Nella media"
  emoji: "💪"
  message: "Stabile e costante, continua"
